Create a `_config.cfg`

```
[setup]
home_path = /path/to/folder
debug_mode = False
twitter_username = USER

[twitter_api]
consumer_key = CONSUMER KET
consumer_secret = consumer_secret
access_token_key = access_token_key
access_token_secret = access_token_secret

[tweets]
last_mention_id = 0

[tweet_text]
photo = ["Looking good [[user]]!", "Very nice shoes [[user]]!"]
video = ["Here is your #graffcam short [[user]]", "Enjoy your film [[user]]"]
```
